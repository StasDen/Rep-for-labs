#include <stdio.h>
#include <malloc.h>
#include <math.h>

int main()
{
	int *massif, inputNumber, sizeNumber;
    	printf("Input size: ");
    	scanf("%d", &sizeNumber);
    		
massif = (int*) malloc (sizeNumber * sizeof(int));
  
	for (inputNumber = 0; inputNumber < sizeNumber; inputNumber++)
  {
    printf("a[%d] = ", inputNumber);
    scanf("%d", &massif[inputNumber]);
  }
  
  
  int element, nextElement, tmp;
  tmp = 0;
 
	for(element=0; element < sizeNumber-1; element++)
	{
    nextElement = element + 1;
    	if(massif[element] >= massif[nextElement])
    	{		
        printf("False");
        	tmp++;
        		break;
   		}
	}
 
		if(tmp != 1)
		{
  			printf("True");
		}
}
